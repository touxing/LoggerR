打开控制台查看日志
控制台输入:
Logger.Logger.download()
下载日志文件
解压日志文件:
import pako from 'pako'; import fs from 'node:fs/promises'; import path from 'node:path'; async function main() { let log = '7da70d38-e778-4851-9741-6542a206b2e1' // 日志存放目录 let distance = `C:\\Users\\xxxx\\Downloads\\${log}` fs.readFile(path.normalize(distance)).then(data => { const inflated = pako.inflate(data); fs.writeFile(`./log/${log}.log`, inflated).then(() => { console.log(`output log file ${log}.log done`); }) }).catch(err => { console.error('解压失败',err); }) } main()
